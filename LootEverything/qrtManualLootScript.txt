scn qrtManualLootScript

int bIsLootRoomPressed

int deleteResult
int isPressOpenMainContainer
int bIsPressManualLoot

Begin GameMode
	if bIsPressManualLoot != IsKeyPressed qrtGlobalMCMVariable.manualLootHotkey
		set bIsPressManualLoot to IsKeyPressed qrtGlobalMCMVariable.manualLootHotkey
		if bIsPressManualLoot
		else
		;按钮放开了
			set isPressOpenMainContainer to 1
		endif
	endif
	if isPressOpenMainContainer==1
		set isPressOpenMainContainer to 0
		messageex "loot now"
		call qrtLootMain 0
	endif
End
